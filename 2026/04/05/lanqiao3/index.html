str.replace(pattern, replacement)

  1. pattern- 要查找的模式
    可以是字符串（只替换第一个匹配项）
    也可以是正则表达式（可替换所有匹配项）

  2. replacement- 替换的内容
    可以是字符串
    也可以是函数（用于更复杂的替换逻辑）

  3. 交换顺序
    let str = “张三,李四”;
    str.replace(/(.+),(.+)/, ‘$2,$1’); // “李四,张三”
    (.+) , (.+)
    ↑ ↑ ↑ ↑ ↑
    │ │ │ │ └── 匹配任意字符1次或多次
    │ │ │ └─── 第2个捕获分组开始
    │ │ └────── 匹配逗号字符
    │ └────── 匹配任意字符1次或多次
    └────── 第1个捕获分组开始

  4. 重新格式化

nodeMap的作用是：

proxy 代理

在文件 age.js 中，有一个 person 对象，当它的 age 属性值改变时需要对新值做判断，将其永远控制在 0 - 150 之间。具体需求如下：
如果新属性值在 0 -150 之间（包含 0 和 150），则直接更新。
如果新属性值小于 0，则属性值更新为 0。
如果新属性值大于 150，则属性值更新为 150。
// 请不要更改这个对象里面的内容

let person = {
age: 0,
};

// 使用 Proxy 创建一个代理对象
const personProxy = new Proxy(person, {
set(target, property, value) {
// 只拦截 age 属性的设置操作
if (property === ‘age’) {
// 验证逻辑
if (value < 0) {
target[property] = 0;
} else if (value > 150) {
target[property] = 150;
} else {
target[property] = value;
}
return true; // 表示设置成功
}
// 对于其他属性，直接设置
target[property] = value;
return true;
},

get(target, property) {
return target[property];
}
});

module.exports = personProxy; // 注意：这里导出的是代理对象
